初中刚刚学会堆排序时候写的可爱的程序=-=

```pascal


var
  i,j,k,l,n,x1,x2,y1,y2,x,y,min,r:longint;
  a:array[1..100000,1..2] of longint;
  b,c,d:array[0..100001] of longint;
begin
  read(x1,y1,x2,y2);
  read(n);
  min:=maxlongint;
  for i:=1 to n do
    begin
      read(x,y);
      a[i,1]:=sqr(x-x1)+sqr(y-y1);
      a[i,2]:=sqr(x-x2)+sqr(y-y2);
    end;
  fillchar(b,sizeof(b),-1);
  for i:=1 to n do
    begin
      b[i]:=a[i,1];
      l:=i;
      while  (b[l]>b[l div 2])and (l>1)  do
        begin
          k:=b[l] ;
          b[l]:=b[l div 2];
          b[l div 2]:=k;
          k:=a[l,2];
          a[l,2]:=a[l div 2,2];
          a[l div 2,2]:=k;
          l:=l div 2 ;
        end;
    end;
  for i:=1 to n do
    begin
      l:=1;
      c[i]:=b[1];
      d[i]:=a[1,2];
      while (l*2<=n)  do
        if(b[l*2]>=b[l*2+1]) then begin  b[l]:=b[l*2]; a[l,2]:=a[l*2,2];  l:=l*2; end
        else begin b[l]:=b[l*2+1];a[l,2]:=a[l*2+1,2]; l:=l*2+1; end;
      b[l]:=-1;
      a[l,2]:=-1;
     end;
   for i:=1 to n do
    begin
      if(c[i]+r<min) then min :=c[i]+r;
      if(r<d[i]) then r:=d[i];
    end;
    write(min);
  end.
```
